# spring study git 추가 방법

# 1. git 로컬 리파지토리 생성
git init

# 2. git 업로드 할 파일 추가
# ex) add . 은 전체 파일
git add .

# 3. git 로컬 커밋
git commit -m "메세지 입력"

# 4. git 브랜치 생성
git branch -M main

# 5 . git 리모트 연동
# git remote add origin [url]
git remote add origin https://github.com/dpravity/springstudy.git


# 6. git push
git push -u origin main
